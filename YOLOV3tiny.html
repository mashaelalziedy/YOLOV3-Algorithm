# Read the image and find the bounding boxes with their labels and confidence scores
boxes=[]
confidences=[]
class_ids=[]
for output in layerOutput:
    for detection in output:
        # The produced vector will contain the coordinates at the first 4 elemnts of the score array, so they must be excluded.
        score = detection[5:] 
        class_id=np.argmax(score)
        confidence=score[class_id]
        # Take only the ground boxes that best fit the objects
        if confidence > 0.7: # 0.7 is the threshold
            center_x=int(detection[0]*width)
            center_y=int(detection[1]*height)
            w=int(detection[2]*width)
            h=int(detection[3]*height)
            # Compute the center value
            x=int(center_x - w/2)
            y=int(center_y - h/2)
            # Fill the bounding boxes values
            boxes.append([x,y,w,h])
            confidences.append(float(confidence))
            class_ids.append(class_id)

    x,y,w,h = boxes[i]
    label=str(classes[class_ids[i]])
    confi=str(round(confidences[i], 2))
    cv2.rectangle(image, (x,y),(x+w, y+h), (255,0,0), 1)
    cv2.putText(image, label+" "+confi, (x,y+20), font, 1, (255, 0, 0), 1)
